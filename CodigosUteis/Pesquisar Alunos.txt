
unit UnitPesquisaAlunos;

interface

uses
  Winapi.Windows, Winapi.Messages, System.SysUtils, System.Variants, System.Classes, Vcl.Graphics,
  Vcl.Controls, Vcl.Forms, Vcl.Dialogs, Data.DB, Vcl.Grids, Vcl.DBGrids,
  Vcl.StdCtrls, Vcl.Buttons, Vcl.ExtCtrls;

type
  TFormPesquisaAlunos = class(TForm)
    Label1: TLabel;
    Edit1: TEdit;
    BitBtn1: TBitBtn;
    DBGrid1: TDBGrid;
    RadioGroup1: TRadioGroup;
    procedure BitBtn1Click(Sender: TObject);
  private
    { Private declarations }
  public
    { Public declarations }
  end;

var
  FormPesquisaAlunos: TFormPesquisaAlunos;

implementation

{$R *.dfm}

uses UnitDM;

procedure TFormPesquisaAlunos.BitBtn1Click(Sender: TObject);
begin
   DM.FDQAlunos.Close;

   if RadioGroup1.ItemIndex = 0 then //pesquisar alunos
      DM.FDQAlunos.SQL[4] :=  'AND Nome_Aluno like ' + '''%' + Edit1.Text + '%'''
   else
      DM.FDQAlunos.SQL[4] :=  'AND NomeCurso like ' + '''%' + Edit1.Text + '%''';

   DM.FDQAlunos.Open;
end;

end.
