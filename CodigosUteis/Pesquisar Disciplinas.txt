
unit UnitPesquisaDisciplinas;

interface

uses
  Winapi.Windows, Winapi.Messages, System.SysUtils, System.Variants, System.Classes, Vcl.Graphics,
  Vcl.Controls, Vcl.Forms, Vcl.Dialogs, Vcl.StdCtrls, Vcl.Buttons, Data.DB,
  Vcl.Grids, Vcl.DBGrids;

type
  TFormPesquisaDisciplinas = class(TForm)
    Label1: TLabel;
    Edit1: TEdit;
    BitBtn1: TBitBtn;
    DBGrid1: TDBGrid;
    procedure BitBtn1Click(Sender: TObject);
    procedure FormShow(Sender: TObject);
    procedure FormClose(Sender: TObject; var Action: TCloseAction);
    procedure DBGrid1DblClick(Sender: TObject);
  private
    { Private declarations }
  public
    { Public declarations }
  end;

var
  FormPesquisaDisciplinas: TFormPesquisaDisciplinas;

implementation

{$R *.dfm}

uses UnitDM, UnitDisciplinas;

procedure TFormPesquisaDisciplinas.BitBtn1Click(Sender: TObject);
begin

  DM.FDQDisciplinas.Locate('NomeDisciplina', Edit1.Text, [loPartialKey, loCaseInsensitive]);

end;

procedure TFormPesquisaDisciplinas.DBGrid1DblClick(Sender: TObject);
begin
   DM.FDTDisciplinas.Locate('NomeDisciplina', DM.FDQDisciplinas.FieldbyName('NomeDisciplina').Value, []);
   FormDisciplinas.ShowModal;
end;

procedure TFormPesquisaDisciplinas.FormClose(Sender: TObject;
  var Action: TCloseAction);
begin
  DM.FDQDisciplinas.Close;
end;

procedure TFormPesquisaDisciplinas.FormShow(Sender: TObject);
begin
  DM.FDQDisciplinas.Open;
end;

end.
