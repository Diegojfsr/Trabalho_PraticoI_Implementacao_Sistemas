
unit UnitPesquisaCursos;

interface

uses
  Winapi.Windows, Winapi.Messages, System.SysUtils, System.Variants, System.Classes, Vcl.Graphics,
  Vcl.Controls, Vcl.Forms, Vcl.Dialogs, Data.DB, Vcl.Grids, Vcl.DBGrids,
  Vcl.StdCtrls, Vcl.ExtCtrls;

type
  TFormPesquisaCursos = class(TForm)
    RadioGroup1: TRadioGroup;
    Label1: TLabel;
    Edit1: TEdit;
    Button1: TButton;
    DBGrid1: TDBGrid;
    procedure Button1Click(Sender: TObject);
    procedure Edit1Enter(Sender: TObject);
  private
    { Private declarations }
  public
    { Public declarations }
  end;

var
  FormPesquisaCursos: TFormPesquisaCursos;

implementation

{$R *.dfm}

uses UnitDM;

procedure TFormPesquisaCursos.Button1Click(Sender: TObject);
begin
  DM.FDQCursos.Close;
  DM.FDQCursos.Open;
  if RadioGroup1.ItemIndex = 0 then //usuario escolheu código
  begin
     try
        StrToInt(Edit1.Text);
     except
        raise Exception.Create('Código do Curso Inválido.');
     end;
     DM.FDQCursos.Filter := 'idCurso = ' + edit1.Text;
     DM.FDQCursos.Filtered := True; //ativa o filtro
  end
  else  //usuario escolheu nome do curso
  begin
     DM.FDQCursos.Filter := 'NomeCurso = ' + quotedstr(edit1.Text);
     DM.FDQCursos.Filtered := True; //ativa o filtro
  end;
end;

procedure TFormPesquisaCursos.Edit1Enter(Sender: TObject);
begin
  if RadioGroup1.ItemIndex = 0 then
     Edit1.Hint := 'Entre com valor numérico';
end;

end.
